\fbox{31}\text{Calculons I = } $\int_0^1 2^x \cdot 3^{-x} \cdot 4^x \cdot 5^{-x}\cdot dx$ : \\
\text{Je ne vais rien vous apprendre mais : } $a^{-x} = \frac{1}{a^x}$ \text{et} $a^x \cdot b^x = (a \cdot b)^x $\\
\text{Ainsi I = } $\int_0^1 2^x \cdot \left( \frac{1}{3} \right)^x \cdot 4^x \cdot \left( \frac{1}{5} \right)^x\cdot dx$\\
\text{soit I = } $\int_0^1 \left( \frac{8}{3} \right)^x \cdot \left( \frac{1}{5} \right)^x\cdot dx$ 
\text{voire même I = } $\int_0^1 \left( \frac{8}{15} \right)^x \cdot dx$ \\
\underline{\text{Comment intégrer ça? \textit{On va utiliser la forme $e^{ln(..)}$}}:}\\
\text{Ainsi} : $a^x = e^{x \cdot ln(a)}$ \text{et} $x \longrightarrow e^{x \cdot ln(a)} $ \text{se primitive en :} $ x \longrightarrow \frac{e^{x \cdot ln(a)}}{ln(a)}$\\
\text{Maintenant, il ne reste plus grand chose, on a :} I = $\left[ \frac{\left( \frac{8}{15} \right)^x}{ln(\frac{8}{15})} \ \right]_0^1$ = $\left[ \frac{8^x}{ln(\frac{8}{15}) \cdot 15^x} \ \right]_0^1$ 
\text{Après calcul, on obtient :} I = $\frac{8^1}{ln(\frac{8}{15}) \cdot 15^1} - \frac{8^0}{ln(\frac{8}{15}) \cdot 15^0} = - \frac{7}{ln(\frac{8}{15}) \cdot 15}$ \\
